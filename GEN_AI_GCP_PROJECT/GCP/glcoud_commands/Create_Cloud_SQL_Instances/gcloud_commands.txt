Enable API
==========


Enable APIS:
============

gcloud services enable compute.googleapis.com \
    servicenetworking.googleapis.com \
    sqladmin.googleapis.com \
    compute.googleapis.com \
    containerregistry.googleapis.com \
    artifactregistry.googleapis.com \
    identitytoolkit.googleapis.com \
    run.googleapis.com \
    vpcaccess.googleapis.com \
    cloudbuild.googleapis.com


Reserve Private Service IP for Cloud SQL
========================================

gcloud compute addresses create google-managed-services-custom-vpc \
  --global \
  --purpose=VPC_PEERING \
  --prefix-length=24 \
  --network=custom-vpc \
  --addresses=10.10.0.0 \
  --description="Peering range for Cloud SQL" \
  --project=YOUR_PROJECT_ID


Create VPC Peering to Google services
=====================================

gcloud services vpc-peerings connect \
  --service=servicenetworking.googleapis.com \
  --ranges=google-managed-services-custom-vpc \
  --network=custom-vpc \
  --project=YOUR_PROJECT_ID


Create SQL instance
===================

gcloud sql instances create vector-db-instance \
  --project=YOUR_PROJECT_ID \
  --database-version=POSTGRES_18 \
  --cpu=2 \
  --memory=4GB \
  --region=us-central1 \
  --network=projects/YOUR_PROJECT_ID/global/networks/custom-vpc \
  --no-assign-ip \
  --enable-private-path-for-google-cloud-services \
  --root-password=test \
  --tier=db-custom-2-4096 \
  --availability-type=regional \
  --storage-type=SSD \
  --storage-size=20 \
  --storage-auto-increase \
  --backup-start-time=02:00 \
  --enable-backup \
  --retained-backups=7 \
  --retention-unit=TIME




Inside PSQL
===========

CREATE DATABASE vector_db;
\c vector_db
CREATE EXTENSION vector;

Enable the pgvector extension
=============================

gcloud sql connect vector-db-instance --user=postgres --project=YOUR_PROJECT_ID

Verify pgvector version
-----------------------

SELECT extname, extversion FROM pg_extension WHERE extname = 'vector';


Requirements
------------

Make sure:

Your Cloud Run service account has Cloud SQL Client IAM permission.

You have created a VPC connector (cloud-run-vpc-connector) in the same region.

Create VPC connector:
---------------------

gcloud compute networks vpc-access connectors create cloud-run-vpc-connector \
  --region=us-central1 \
  --network=custom-vpc \
  --range=10.8.0.0/28
